% <trunk>

\let\oldsubsection\subsection
\def\subsection#1
{
  \def\tmpa{}
  \def\tmpb{#1}
  \ifx\tmpa\tmpb
  {
    \edef\addcontentsline##1##2##3{}
    \oldsubsection{}
  }
  \else
    \oldsubsection{\tmpb}
  \fi
}

% </trunk>
